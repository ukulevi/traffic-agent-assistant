\section{Tổng quan Tầng Điều phối}

Tầng 4 là ``\textbf{bộ não}'' của hệ thống STWI --- chuyển hóa kết quả từ ba tầng dưới thành đề xuất hành động cụ thể, có căn cứ pháp lý và đã được kiểm tra an toàn. Tầng này xây dựng trên nền tảng LangChain/CrewAI và triển khai ba cơ chế quan trọng:

\begin{itemize}[itemsep=3pt]
  \item \textbf{Task Decomposition:} Phân rã câu hỏi What-if thành các subtask song song.
  \item \textbf{Multi-Agent Coordination:} Điều phối 3 tác tử con hoạt động đồng thời.
  \item \textbf{CF-VLA Safety Loop:} Phản biện phản thực tế bắt buộc trước khi xuất đề xuất.
\end{itemize}

\subsection{Pseudo-code CF-VLA Engine}

\begin{algorithm}[H]
\caption{CF-VLA Counterfactual Safety Loop}
\label{alg:cfvla}
\Input{Đề xuất sơ bộ $a_0$, danh sách nút lân cận $\mathcal{N}(v)$, ADE model $f_\theta$}
\Output{Đề xuất an toàn $a^*$ hoặc cảnh báo}
\BlankLine
$\text{max\_iter} \leftarrow 3$\;
$a \leftarrow a_0$\;
\BlankLine
\For{$k \leftarrow 1$ \KwTo $\text{max\_iter}$}{
  \tcc{Bước 2: Counterfactual Reflection}
  \ForEach{nút $v_j \in \mathcal{N}(v)$}{
    $\mathbf{c}_{\text{surge}} \leftarrow \text{ApplySurge}(a, v_j, \text{surge\_factor}=1.4)$\;
    $\hat{\mathbf{Y}}_j \leftarrow f_\theta(\mathbf{X}_{\text{current}}, \mathbf{c}_{\text{surge}})$\;
    $\text{vc}_j \leftarrow \text{ComputeVC}(\hat{\mathbf{Y}}_j, \text{capacity}[v_j])$\;
  }
  \BlankLine
  \eIf{$\max_{v_j \in \mathcal{N}(v)} \text{vc}_j \leq 0.9$}{
    $a^* \leftarrow a$\;
    \textbf{return} $a^*$, status=``SAFE''\;
  }{
    \tcc{Bước 3b: Hiệu chỉnh phương án}
    $v_{\text{worst}} \leftarrow \arg\max_{v_j} \text{vc}_j$\;
    $a \leftarrow \text{CorrectAction}(a, v_{\text{worst}}, \text{vc}[v_{\text{worst}}])$\;
    \Comment{Ví dụ: giảm 20\% phân luồng sang đường bị quá tải}
  }
}
\tcc{Het so lan lap, van khong dat nguong an toan}
\textbf{return} $a$, status=``[CANH BAO - CHUA KIEM CHUNG]''\;
\end{algorithm}
