\section{Đồ thị Giao thông và Biểu diễn Không gian}

\subsection{Xây dựng Ma trận Kề}

Mạng chức năng MVP 20 node được biểu diễn bằng đồ thị $\mathcal{G} = (\mathcal{V}, \mathcal{E}, \mathbf{W})$. Ma trận trọng số $\mathbf{W}$ được xây dựng từ dữ liệu bản đồ thực (OpenStreetMap):

\begin{equation}
  W_{ij} = \begin{cases}
    \exp\!\left(-\dfrac{d_{ij}^2}{2\sigma^2}\right) & \text{nếu } (v_i, v_j) \in \mathcal{E} \text{ và } d_{ij} \leq d_{\max} \\
    0 & \text{ngược lại}
  \end{cases}
  \label{eq:weight_matrix}
\end{equation}

với $d_{ij}$ là khoảng cách đường phố (km) giữa nút $i$ và $j$, $\sigma = 1{,}5$ km (bán kính ảnh hưởng trực tiếp), $d_{\max} = 3$ km.

\subsection{Chi tiết Lớp GCN}

Với $L=2$ lớp GCN, quá trình cập nhật đặc trưng diễn ra như sau:

\begin{equation}
  \mathbf{H}^{(1)} = \text{ReLU}\!\left(\hat{\mathbf{A}}\mathbf{X}_t\mathbf{W}^{(0)}\right), \qquad
  \mathbf{H}^{(2)} = \text{ReLU}\!\left(\hat{\mathbf{A}}\mathbf{H}^{(1)}\mathbf{W}^{(1)}\right)
  \label{eq:gcn_layers}
\end{equation}

trong đó $\hat{\mathbf{A}} = \tilde{\mathbf{D}}^{-1/2}\tilde{\mathbf{A}}\tilde{\mathbf{D}}^{-1/2}$ là ma trận kề chuẩn hóa, $\mathbf{X}_t \in \mathbb{R}^{N \times 16}$ là đặc trưng tại bước thời gian $t$.

\textbf{Số tham số của module GCN:}
\begin{align}
  |\theta_{\text{GCN-1}}| &= 16 \times 32 = 512 \text{ tham số} \notag\\
  |\theta_{\text{GCN-2}}| &= 32 \times 64 = 2.048 \text{ tham số} \notag
\end{align}

\subsection{Hàm Loss và Huấn luyện}

Hàm mất mát kết hợp RMSE và MAE:

\begin{equation}
  \mathcal{L} = \alpha \cdot \text{RMSE}(\hat{\mathbf{Y}}, \mathbf{Y}) + (1-\alpha) \cdot \text{MAE}(\hat{\mathbf{Y}}, \mathbf{Y})
  \label{eq:loss}
\end{equation}

với $\alpha = 0{,}7$ để nhấn mạnh các sai lệch lớn (ùn tắc nặng). Bộ tối ưu Adam với learning rate $3 \times 10^{-4}$, warmup 5 epoch, cosine decay.

\begin{lstlisting}[caption={Huấn luyện GCN–LSTM với PyTorch (trích)},label={lst:training}]
import torch
import torch.nn as nn
from torch_geometric.nn import GCNConv

class GCNLSTMModel(nn.Module):
    def __init__(self, num_nodes: int, in_features: int = 16,
                 gcn_hidden: int = 64, lstm_hidden: int = 128,
                 forecast_steps: int = 6, out_features: int = 2):
        super().__init__()
        self.N = num_nodes
        # Spatial Module
        self.gcn1 = GCNConv(in_features, 32)
        self.gcn2 = GCNConv(32, gcn_hidden)
        # Temporal Module
        self.lstm = nn.LSTM(
            input_size=num_nodes * gcn_hidden,
            hidden_size=lstm_hidden, num_layers=3,
            batch_first=True, dropout=0.2
        )
        # Output projection
        self.head = nn.Linear(
            lstm_hidden, forecast_steps * num_nodes * out_features
        )
        self.forecast_steps = forecast_steps
        self.out_features = out_features

    def forward(self, x: torch.Tensor, edge_index: torch.Tensor,
                edge_weight: torch.Tensor) -> torch.Tensor:
        B, T, N, F = x.shape  # [Batch, 12, N_nodes, 16]

        # 1. GCN tren tung buoc thoi gian
        spatial_feats = []
        for t in range(T):
            h = torch.relu(self.gcn1(x[:, t], edge_index, edge_weight))
            h = torch.relu(self.gcn2(h, edge_index, edge_weight))
            spatial_feats.append(h)               # [B, N, 64]
        spatial_seq = torch.stack(spatial_feats, dim=1)  # [B, 12, N, 64]

        # 2. Flatten nodes -> sequence dau vao LSTM
        seq = spatial_seq.view(B, T, -1)          # [B, 12, N*64]

        # 3. Stacked LSTM
        lstm_out, _ = self.lstm(seq)              # [B, 12, 128]
        last_hidden = lstm_out[:, -1, :]          # [B, 128]

        # 4. Dau ra du bao
        out = self.head(last_hidden)
        return out.view(B, self.forecast_steps, N, self.out_features)
\end{lstlisting}

\section{Surrogate Ensemble cho Kịch bản What-if}

GCN--LSTM dự báo trạng thái baseline nếu không có can thiệp. Surrogate ensemble nhận baseline cùng \textit{IncidentVector} và xấp xỉ kết quả SUMO cho một hành động giả định.

\subsection{Tập dữ liệu SUMO offline}

\subsection{Kiến trúc và calibration}

Ba sub-model MLP, CNN-1D và light Transformer được huấn luyện độc lập trên cùng schema. Weighted mean dùng validation performance; variance giữa model chỉ là tín hiệu thô và phải được calibration bằng held-out data.

\begin{itemize}[itemsep=3pt]
  \item Output theo node/horizon gồm volume, speed, V/C và uncertainty; summary gồm network delay, clearance time và max V/C.
  \item Threshold uncertainty/OOD được cố định trước test.
  \item Case lịch sử chỉ dùng làm evidence online hoặc dữ liệu huấn luyện offline; không blend trực tiếp vào input.
  \item OOD, uncertainty cao hoặc thiếu case xác nhận dẫn đến \texttt{needs\_review}.
\end{itemize}

\subsection{Benchmark Tốc độ}

\begin{tcolorbox}[dangerbox]
\textbf{Ràng buộc cứng về hiệu năng (QC-2):} Nếu $P_{99}$ TTP vượt 500ms, Surrogate Model \textbf{phải được tối ưu lại} theo thứ tự ưu tiên: (1) Quantization INT8, (2) Model Pruning 20--30\%, (3) Giảm ensemble size từ $N$ xuống $N/2$, (4) Caching kết quả với similar vector hash.
\end{tcolorbox}

\begin{lstlisting}[caption={Benchmark TTP với percentile đo lường},label={lst:benchmark}]
import time
import numpy as np
from typing import List

def benchmark_surrogate_ttp(model, test_inputs: List, n_runs: int = 1000) -> dict:
    """
    Benchmark Time-to-Prediction cua Surrogate Ensemble.
    Ket qua phai dat: P99 < 500ms, P50 < 150ms.
    """
    latencies_ms = []

    for inp in test_inputs[:n_runs]:
        t0 = time.perf_counter()
        _ = model.predict(inp)  # Forward pass
        t1 = time.perf_counter()
        latencies_ms.append((t1 - t0) * 1000)

    latencies = np.array(latencies_ms)
    results = {
        'P50_ms':  float(np.percentile(latencies, 50)),
        'P95_ms':  float(np.percentile(latencies, 95)),
        'P99_ms':  float(np.percentile(latencies, 99)),
        'mean_ms': float(np.mean(latencies)),
        'max_ms':  float(np.max(latencies)),
    }

    # Kiem tra rang buoc cung QC-2
    if results['P99_ms'] > 500:
        raise RuntimeError(
            f"[QC-2 VIOLATION] TTP P99 = {results['P99_ms']:.1f}ms > 500ms. "
            "Can toi uu lai model truoc khi deploy."
        )

    print(f"TTP Benchmark: P50={results['P50_ms']:.1f}ms | "
          f"P99={results['P99_ms']:.1f}ms | Max={results['max_ms']:.1f}ms")
    return results
\end{lstlisting}
