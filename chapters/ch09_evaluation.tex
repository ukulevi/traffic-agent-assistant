\section{Kiểm thử Hiệu năng Hệ thống}

\subsection{Load Test Configuration}

\begin{lstlisting}[caption={Cấu hình kiểm thử tải với locust},label={lst:loadtest}]
# locustfile.py - STWI Load Test Configuration
from locust import HttpUser, task, between
import random, json

WHAT_IF_SCENARIOS = [
    "Tai nan o nga tu Nguyen Hue - Le Loi...",
    "Ngap ut tai duong Nguyen Van Cu...",
    "Dong duong Bach Dang de bao duong...",
]

class STWIOperator(HttpUser):
    wait_time = between(1, 5)  # Giay cho giua cac request

    @task(weight=3)
    def submit_what_if(self):
        payload = {
            "scenario": random.choice(WHAT_IF_SCENARIOS),
            "constraints": {
                "max_computation_time_ms": 180000,
                "priority": "high",
                "cf_vla_enabled": True
            }
        }
        with self.client.post(
            "/api/v1/agent/what-if",
            json=payload,
            timeout=200,  # 200 giay timeout
            catch_response=True
        ) as resp:
            if resp.status_code == 200:
                data = resp.json()
                # Kiem tra SLA: E2E < 3 phut = 180,000ms
                if data.get("latency_ms", 999999) > 180000:
                    resp.failure(f"SLA violated: {data['latency_ms']}ms")
                # Kiem tra No-Hallucination
                if "legal_grounding" not in data:
                    resp.failure("Missing legal_grounding field")
            else:
                resp.failure(f"HTTP {resp.status_code}")
\end{lstlisting}
