\section{Data Contract --- Pydantic Models}

\begin{lstlisting}[caption={Data Contract định nghĩa Tensor shape (bất biến)},label={lst:pydantic}]
from pydantic import BaseModel, field_validator
from typing import Literal
import torch

class TensorContract(BaseModel):
    """Hop dong bat bien cho Tensor dau vao Tầng 2.
    Moi thay doi shape phai qua HĐ Kien truc phe duyet (QC-3).
    """
    batch_size:  int = 32
    time_steps:  int = 12   # Bat buoc: 60 phut / 5 phut = 12 buoc
    num_features: int = 14  # Bat buoc: 3 GT + 5 KT + 3 KK + 3 PT

    @field_validator('time_steps')
    def time_steps_must_be_12(cls, v):
        if v != 12:
            raise ValueError(f"time_steps phai la 12, nhan {v}. "
                           "Thay doi can phe duyet cua HĐ Kien truc.")
        return v

    @field_validator('num_features')
    def features_must_be_14(cls, v):
        if v != 14:
            raise ValueError(f"num_features phai la 14, nhan {v}.")
        return v

class IncidentVector(BaseModel):
    """Dac ta Vector Su co cho ADE Surrogate Model."""
    incident_node_id: str
    incident_type: Literal['accident', 'flood', 'closure', 'event']
    severity: float          # [0.0, 1.0]
    lanes_closed: int        # [0, 4]
    duration_min: float
    affected_edges: list[str]
    time_of_day: float       # [0.0, 1.0] = gio/24
    weather_flag: Literal['normal', 'rain', 'flood']

class WhatIfRequest(BaseModel):
    """Schema request cho API What-if."""
    scenario: str            # Min length 10 chars
    operator_id: str
    max_computation_time_ms: int = 180000
    priority: Literal['low', 'normal', 'high'] = 'normal'
    cf_vla_enabled: bool = True

    @field_validator('cf_vla_enabled')
    def cf_vla_cannot_be_disabled(cls, v):
        if not v:
            raise ValueError("CF-VLA khong the tat (QC-5). "
                           "He thong tu choi moi request voi cf_vla_enabled=False.")
        return v
\end{lstlisting}
