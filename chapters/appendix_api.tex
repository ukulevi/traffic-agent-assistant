\section{Mô tả Chi tiết Request/Response}

\subsection{POST /api/v1/agent/what-if}

\textbf{Request Schema (JSON):}

\begin{lstlisting}[caption={Request schema cho What-if API},label={lst:req_schema}]
{
  "scenario":    string,   // Bat buoc: Mo ta kich ban (tieng Viet/Anh)
  "operator_id": string,   // Bat buoc: ID nguoi dieu hanh
  "constraints": {
    "max_computation_time_ms": integer,  // Default: 180000 (3 phut)
    "priority": "low"|"normal"|"high",   // Default: "normal"
    "cf_vla_enabled": boolean,           // Default: true (KHONG the false)
    "nodes_of_interest": [string],       // Tuy chon: danh sach node ID
  }
}
\end{lstlisting}

\textbf{Response Schema (JSON):}

\begin{lstlisting}[caption={Response schema đầy đủ với 7 trường bắt buộc},label={lst:resp_schema}]
{
  "status":             "success"|"error"|"timeout",
  "scenario_id":        string,          // UUID cua kich ban
  "recommended_action": {                // Bat buoc
    "primary":          string,
    "secondary":        string,
    "signal_timing":    object,
    "confidence":       float            // 0.0 - 1.0
  },
  "simulation_metrics": {               // Bat buoc
    "affected_nodes":   [string],
    "node_vc_ratios":   {string: float},
    "clearance_time_mins": integer,
    "max_vc_after":     float
  },
  "legal_grounding": {                  // Bat buoc (QC-7)
    "sop_ref":          string,
    "law_ref":          string,
    "source_chunks":    [string],
    "confidence":       float
  },
  "cf_vla_reflection": {               // Bat buoc
    "iterations":       integer,
    "initial_rejected": boolean,
    "rejection_reason": string|null,
    "corrections_applied": [string]
  },
  "latency_ms":         integer,        // Bat buoc
  "timestamp":          string,         // ISO 8601
  "warning_label":      string|null     // "[CANH BAO...]" neu ngoai SOP
}
\end{lstlisting}
